# 使用koa-body以支持请求的文件类型

// 解析请求体
app.use(koaBody({
    // 支持文件类型： multipart/form-data
    multipart: true,
    formidable: {
        // 文件路径
        uploadDir: path.join(__dirname, '/public/uploads'),
        // 保留文件后缀
        keepExtensions: true,
    }
}));

# 接口实现

upload(ctx) {
    const file = ctx.request.files.file;
    const basename = path.basename(file.path);
    ctx.body = {
        url: `${ctx.origin}/uploads/${basename}`
    }
}
